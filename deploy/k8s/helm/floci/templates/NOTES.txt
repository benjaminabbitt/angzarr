Test with AWS CLI:
  export AWS_ACCESS_KEY_ID=test
  export AWS_SECRET_ACCESS_KEY=test
  export AWS_DEFAULT_REGION={{ .Values.config.defaultRegion }}

  # List S3 buckets
  aws --endpoint-url=http://localhost:{{ .Values.service.port }} s3 ls

  # Create SQS queue
  aws --endpoint-url=http://localhost:{{ .Values.service.port }} sqs create-queue --queue-name test-queue

  # List DynamoDB tables
  aws --endpoint-url=http://localhost:{{ .Values.service.port }} dynamodb list-tables
