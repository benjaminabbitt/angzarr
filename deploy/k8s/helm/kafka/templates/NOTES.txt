Check cluster status:
  kubectl get kafka {{ include "angzarr-mq-kafka.fullname" . }} -n {{ .Release.Namespace }}
  kubectl get kafkanodepools -l strimzi.io/cluster={{ include "angzarr-mq-kafka.fullname" . }} -n {{ .Release.Namespace }}

View logs:
  # Broker logs
  kubectl logs -l strimzi.io/cluster={{ include "angzarr-mq-kafka.fullname" . }},strimzi.io/kind=Kafka -n {{ .Release.Namespace }}

List topics:
  kubectl exec -it {{ include "angzarr-mq-kafka.fullname" . }}-broker-0 -n {{ .Release.Namespace }} -- \
    bin/kafka-topics.sh --bootstrap-server localhost:9092 --list

Create topic:
  kubectl apply -f - <<EOF
  apiVersion: kafka.strimzi.io/v1beta2
  kind: KafkaTopic
  metadata:
    name: my-topic
    namespace: {{ .Release.Namespace }}
    labels:
      strimzi.io/cluster: {{ include "angzarr-mq-kafka.fullname" . }}
  spec:
    partitions: 3
    replicas: 1
  EOF
